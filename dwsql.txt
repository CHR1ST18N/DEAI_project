CREATE TABLE DIM_Product (
    product_number      INT PRIMARY KEY,
    region              VARCHAR(45),
    introduction_date   DATE,
    production_code     INT,
    margin              DECIMAL(10,2),
    product_image       VARCHAR(100),
    language            VARCHAR(10),
    product_name        VARCHAR(100),
    product_type_en     VARCHAR(45),
    product_line_en     VARCHAR(45)
);

CREATE TABLE DIM_Inventory_Levels (
    inventory_levels_code   INT PRIMARY KEY,
    inventory_year          INT,
    inventory_month         INT,
    inventory_count         INT,
    product_number          INT,
    FOREIGN KEY (product_number) REFERENCES DIM_Product(product_number)
);

CREATE TABLE DIM_Product_Forecast (
    product_forecast_code   INT PRIMARY KEY,
    year                    INT,
    month                   INT,
    expected_volume         INT,
    product_number          INT,
    FOREIGN KEY (product_number) REFERENCES DIM_Product(product_number)
);

CREATE TABLE DIM_Month (
    month_id    INT PRIMARY KEY,
    monthnr     INT,
    month       VARCHAR(20),
    quarter     INT,
    year        INT
);

CREATE TABLE DIM_Date (
    date        DATE PRIMARY KEY,
    day         INT,
    month_id    INT,
    FOREIGN KEY (month_id) REFERENCES DIM_Month(month_id)
);

CREATE TABLE FACT_Sales_Target (
    sales_target_code   INT PRIMARY KEY,
    sales_year          INT,
    sales_period        INT,
    retailer_name       VARCHAR(100),
    retailer_code       INT,
    expected_volume     INT,
    sales_target        DECIMAL(10,2),
    sales_staff_code    INT,
    product_number      INT,
    FOREIGN KEY (product_number) REFERENCES DIM_Product(product_number)
);

CREATE TABLE FACT_Returned_Item (
    return_code         INT PRIMARY KEY,
    return_date         DATE,
    return_quantity     INT,
    return_reason_en    VARCHAR(100),
    order_details       INT
);

CREATE TABLE FACT_Order_Details (
    order_detail_code   INT PRIMARY KEY,
    quantity            INT,
    unit_cost           DECIMAL(10,2),
    unit_price          DECIMAL(10,2),
    unit_revenue        DECIMAL(10,2),
    unit_sale_price     DECIMAL(10,2),
    order_header        INT,
    product             INT,
    FOREIGN KEY (product) REFERENCES DIM_Product(product_number)
);

CREATE TABLE FACT_Order_Header (
    order_number            INT PRIMARY KEY,
    retailer_name           VARCHAR(100),
    retailer_contact_code   INT,
    order_date              DATE,
    order_method_en         VARCHAR(45),
    retailer_site           INT,
    sales_staff             INT,
    sales_branch            INT
);

CREATE TABLE DIM_Retailer_Site (
    retailer_site_code  INT PRIMARY KEY,
    retailer_code       INT,
    address1            VARCHAR(100),
    address2            VARCHAR(100),
    city                VARCHAR(100),
    region              VARCHAR(100),
    postal_zone         VARCHAR(45),
    active_indicator    BOOLEAN,
    customer            INT,
    country             INT
);

CREATE TABLE DIM_Customer_Store (
    customer_store_code INT PRIMARY KEY,
    street              VARCHAR(100),
    addition            VARCHAR(100),
    city                VARCHAR(45),
    state               VARCHAR(45),
    zipcode             VARCHAR(45),
    active_indicator    BOOLEAN,
    customer            INT,
    crm_country         INT
);

CREATE TABLE DIM_Customer (
    customer_code           INT PRIMARY KEY,
    company_name            VARCHAR(45),
    customer_type           INT,
    customer_headquarters   INT
);

CREATE TABLE DIM_Customer_Segment (
    segment_code            INT PRIMARY KEY,
    language                VARCHAR(10),
    segment_name            VARCHAR(45),
    segment_description     VARCHAR(100)
);

CREATE TABLE DIM_Customer_Headquarters (
    customer_code_hq    INT PRIMARY KEY,
    customer_name       VARCHAR(100),
    address1            VARCHAR(100),
    address2            VARCHAR(100),
    city                VARCHAR(45),
    region              VARCHAR(45),
    postal_zone         VARCHAR(45),
    phone               VARCHAR(45),
    fax                 VARCHAR(45),
    flag_image          VARCHAR(100),
    country             INT,
    segment_code        INT,
    customer_segment    INT
);

CREATE TABLE DIM_Age_Group (
    age_group_code  INT PRIMARY KEY,
    upper_age       INT,
    lower_age       INT
);

CREATE TABLE FACT_Sales_Demographic (
    demographic_code        INT PRIMARY KEY,
    sales_percent           INT,
    customer_headquarters   INT,
    age_group               INT,
    FOREIGN KEY (customer_headquarters) REFERENCES DIM_Customer_Headquarters(customer_code_hq),
    FOREIGN KEY (age_group)             REFERENCES DIM_Age_Group(age_group_code)
);

CREATE TABLE DIM_Country (
    country_code    INT PRIMARY KEY,
    country         VARCHAR(100),
    language        VARCHAR(10),
    currency_name   VARCHAR(45)
);

CREATE TABLE DIM_Country_CRM (
    country_code        INT PRIMARY KEY,
    country_en          VARCHAR(45),
    flag_image          VARCHAR(100),
    territory_name_en   VARCHAR(45)
);

CREATE TABLE DIM_Sales_Branch (
    sales_branch_code   INT PRIMARY KEY,
    address1            VARCHAR(100),
    address2            VARCHAR(100),
    city                VARCHAR(100),
    region              VARCHAR(100),
    postal_code         VARCHAR(45),
    country             INT,
    FOREIGN KEY (country) REFERENCES DIM_Country(country_code)
);

CREATE TABLE DIM_Sales_Staff (
    sales_staff_code            INT PRIMARY KEY,
    first_name                  VARCHAR(45),
    last_name                   VARCHAR(45),
    position_en                 VARCHAR(100),
    work_phone                  VARCHAR(45),
    extension                   INT,
    fax                         VARCHAR(45),
    email                       VARCHAR(100),
    date_hired                  DATE,
    training_course             VARCHAR(100),
    satisfaction_year           INT,
    satisfaction_description    VARCHAR(100),
    sales_branch                INT,
    manager                     INT,
    FOREIGN KEY (sales_branch) REFERENCES DIM_Sales_Branch(sales_branch_code)
);

CREATE TABLE DIM_Sales_Office (
    sales_office_code   INT PRIMARY KEY,
    street              VARCHAR(100),
    addition            VARCHAR(45),
    city                VARCHAR(100),
    region              VARCHAR(100),
    zipcode             VARCHAR(45),
    country_code        INT,
    FOREIGN KEY (country_code) REFERENCES DIM_Country(country_code)
);

CREATE TABLE DIM_Sales_Representative (
    sales_representative_code   INT PRIMARY KEY,
    first_name                  VARCHAR(45),
    last_name                   VARCHAR(45),
    position_en                 VARCHAR(100),
    work_phone                  VARCHAR(45),
    extension                   INT,
    fax                         VARCHAR(45),
    email                       VARCHAR(100),
    date_hired                  DATE,
    sales_office                INT,
    manager                     INT,
    FOREIGN KEY (sales_office) REFERENCES DIM_Sales_Office(sales_office_code),
    FOREIGN KEY (manager) REFERENCES DIM_Sales_Representative(sales_representative_code)
);
